# Загрузка в словарь из JSON

```1c
&НаКлиенте
Процедура ЗагрузитьJSON(Команда)
	
	Диалог = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
	Диалог.Фильтр = "JSON-файлы (*.json)|*.json";
	Диалог.Заголовок = "Выберите JSON-файл для импорта";
	
	Если Диалог.Выбрать() Тогда
		ЗагрузитьДанныеИзJSONНаСервере(Диалог.ПолноеИмяФайла);
	КонецЕсли;
	
КонецПроцедуры

&НаСервере
Процедура ЗагрузитьДанныеИзJSONНаСервере(ПутьКФайлу)
	
	Чтение = Новый ЧтениеJSON;
	Чтение.ОткрытьФайл(ПутьКФайлу);
	
	// Читаем как массив объектов (самый частый вариант)
	Данные = ПрочитатьJSON(Чтение, Истина); // Истина = читать в Соответствие
	Чтение.Закрыть();
	
	// Если JSON начинается с {"items": [...]} — раскомментируйте строку ниже:
	// Данные = Данные["items"];
	
	Если ТипЗнч(Данные) <> Тип("Массив") Тогда
		Сообщить("JSON должен быть массивом объектов!");
		Возврат;
	КонецЕсли;
	
	Для Каждого СтрокаДанных Из Данные Цикл
		
		Код          = СокрЛП(СтрокаДанных["Код"]);           // или СтрокаДанных.Код
		Наименование = СокрЛП(СтрокаДанных["Наименование"]);
		
		ЭлементСсылка = Справочники.Справочник2.НайтиПоКоду(Код);
		
		Если ЗначениеЗаполнено(ЭлементСсылка) Тогда
			ЭлементОбъект = ЭлементСсылка.ПолучитьОбъект();
		Иначе
			ЭлементОбъект = Справочники.Справочник2.СоздатьЭлемент();
			ЭлементОбъект.Код = Код;
		КонецЕсли;
		
		ЭлементОбъект.Наименование = Наименование;
		// ЭлементОбъект.Артикул = СтрокаДанных["Артикул"];
		
		Попытка
			ЭлементОбъект.Записать();
			Сообщить("Загружена номенклатура: " + Наименование);
		Исключение
			Сообщить("Ошибка: " + ОписаниеОшибки());
		КонецПопытки;
		
	КонецЦикла;
	
	Сообщить("Импорт JSON завершён!");
	
КонецПроцедуры
```
