# Код для общей формы для загрузки данных в словари из внешних файлов

```1c
&НаКлиенте
Процедура ВыборФайла(Команда)
	Диалог = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
    Диалог.Заголовок = "Выберите файл";
    Диалог.Фильтр = "CSV файлы (*.csv)|*.csv|Все файлы (*.*)|*.*";
    Диалог.МножественныйВыбор = Ложь;
    
    Если Диалог.Выбрать() Тогда
        ПутьКФайлу = Диалог.ПолноеИмяФайла;
		//Сообщить(ПутьКФайлу);             
		ЗагрузитьКлиентовНаСервере(ПутьКФайлу);
        // Обработка выбранного файла
	КонецЕсли;
КонецПроцедуры


&НаСервере
Процедура ЗагрузитьКлиентовНаСервере(ПутьКФайлу)
    ТекстовыйДокумент = Новый ТекстовыйДокумент;
    ТекстовыйДокумент.Прочитать(ПутьКФайлу, КодировкаТекста.UTF8);
    
    Разделитель = ";"; // или "," для английского CSV
    
    // Пропускаем заголовок (1 строка)
    Для НомерСтроки = 2 По ТекстовыйДокумент.КоличествоСтрок() Цикл
        СтрокаCSV = ТекстовыйДокумент.ПолучитьСтроку(НомерСтроки);
        ДанныеСтроки = СтрРазделить(СтрокаCSV, Разделитель);
        
        Если ДанныеСтроки.Количество() < 2 Тогда
            Продолжить; // Пустая строка
        КонецЕсли;
        
        СоздатьКлиентаИзСтроки(ДанныеСтроки);
    КонецЦикла;
    
	//Сообщить("Загружено клиентов: " + СчетчикКлиентов);        
КонецПроцедуры


&НаСервере
Процедура СоздатьКлиентаИзСтроки(ДанныеСтроки)
    Клиент = Справочники.Клиенты.СоздатьЭлемент(); // Замените на ваше имя справочника
    
    Клиент.Наименование = СокрЛП(ДанныеСтроки[0]); // ФИО
    Если ДанныеСтроки.Количество() > 1 Тогда
        Клиент.Телефон = СокрЛП(ДанныеСтроки[1]);     // Телефон
    КонецЕсли;
    Если ДанныеСтроки.Количество() > 2 Тогда
        Клиент.email = СокрЛП(ДанныеСтроки[2]);      // Email
    КонецЕсли;
    
    // Проверяем дубль по наименованию
    Запрос = Новый Запрос;
    Запрос.Текст = 
        "ВЫБРАТЬ ПЕРВЫЕ 1
        |    Клиенты.Ссылка
        |ИЗ
        |    Справочник.Клиенты КАК Клиенты
        |ГДЕ
        |    Клиенты.Наименование = &Наименование";
    Запрос.УстановитьПараметр("Наименование", Клиент.Наименование);
    
    Если НЕ Запрос.Выполнить().Пустой() Тогда
        Возврат; // Пропускаем дубль
    КонецЕсли;
    
    Попытка
        Клиент.Записать();
        СчетчикКлиентов = СчетчикКлиентов + 1;
    Исключение
        Сообщить("Ошибка записи клиента " + Клиент.Наименование + ": " + ОписаниеОшибки());
    КонецПопытки;
КонецПроцедуры

```
