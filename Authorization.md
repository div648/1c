# Форма авторизации

```1c
&НаКлиенте
Процедура Войти(Команда)
    Если Не ЗначениеЗаполнено(Логин) Или Не ЗначениеЗаполнено(Пароль) Тогда
        Сообщить("Поля 'Логин' и 'Пароль' обязательны для заполнения!");
        Возврат;
    КонецЕсли;

    Авторизован = ПроверитьАвторизацию(Логин, Пароль);
    Если Авторизован Тогда
        Сообщить("Вы успешно авторизовались");
        // Здесь можно добавить логику: открыть основную форму, установить параметры сеанса и т.д.
        Закрыть();
    Иначе
        Сообщить("Вы ввели неверный логин или пароль. Пожалуйста проверьте ещё раз введенные данные");
    КонецЕсли;
КонецПроцедуры

&НаСервереБезКонтекста
Функция ПроверитьАвторизацию(Логин, Пароль)
    Запрос = Новый Запрос;
    Запрос.Текст = 
    "ВЫБРАТЬ
    |    Пользователи.Ссылка
    |ИЗ
    |    Справочник.Пользователи КАК Пользователи
    |ГДЕ
    |    Пользователи.Логин = &Логин
    |    И Пользователи.Пароль = &Пароль";
    Запрос.УстановитьПараметр("Логин", Логин);
    Запрос.УстановитьПараметр("Пароль", Пароль);
    Результат = Запрос.Выполнить();
    Возврат Не Результат.Пустой();
КонецФункции
```
