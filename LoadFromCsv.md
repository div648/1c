# Загрузка в словарь из CSV

```1c
&НаКлиенте
Процедура ЗагрузитьCSV(Команда)

	Диалог = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
	Диалог.Фильтр = "CSV-файлы (*.csv)|*.csv";
	Диалог.Заголовок = "Выберите CSV-файл для импорта";
		
	Если Диалог.Выбрать() Тогда
		ЗагрузитьДанныеИзCSVНаСервере(Диалог.ПолноеИмяФайла);
	КонецЕсли;
		
КонецПроцедуры

&НаСервере
Процедура ЗагрузитьДанныеИзCSVНаСервере(ПутьКФайлу)
		
	ТД = Новый ТекстовыйДокумент;
	ТД.Прочитать(ПутьКФайлу, КодировкаТекста.UTF8); 
	//ТД.Прочитать(ПутьКФайлу, КодировкаТекста.ANSI); // если кодировка Windows-1251 — КодировкаТекста.ANSI
		
	Для НомерСтроки = 2 По ТД.КоличествоСтрок() Цикл
		
		СтрокаФайла = СокрЛП(ТД.ПолучитьСтроку(НомерСтроки));
		Если ПустаяСтрока(СтрокаФайла) Тогда
			Продолжить;
		КонецЕсли;
		
		// Если запятая — заменить на ","
		МассивЗначений = СтрРазделить(СтрокаФайла, ";", Ложь);
		
		Если МассивЗначений.Количество() < 2 Тогда
			Продолжить;
		КонецЕсли;
		
		Код           = СокрЛП(МассивЗначений[0]);
		Наименование  = СокрЛП(МассивЗначений[1]);
		
		// === Импорт в справочник ===
		ЭлементСсылка = Справочники.Справочник1.НайтиПоКоду(Код);
		
		Если ЗначениеЗаполнено(ЭлементСсылка) Тогда
			ЭлементОбъект = ЭлементСсылка.ПолучитьОбъект();
		Иначе
			ЭлементОбъект = Справочники.Справочник1.СоздатьЭлемент();
			ЭлементОбъект.Код = Код;
		КонецЕсли;
		
		ЭлементОбъект.Наименование = Наименование;
		
		ЭлементОбъект.Артикул = СокрЛП(МассивЗначений[2]);
		
		Попытка
			ЭлементОбъект.Записать();
			Сообщить("Загружена номенклатура: " + Наименование);
		Исключение
			Сообщить("Ошибка при записи " + Наименование + ": " + ОписаниеОшибки());
		КонецПопытки;
		
	КонецЦикла;
		
	Сообщить("Импорт CSV завершён!");
		
КонецПроцедуры
```
