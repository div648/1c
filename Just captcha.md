## Дерево формы

- Капча
  - ГруппаКапча
    - ГруппаКартинки
      - Декорация1
      - Декорация2
      - Декорация3
      - Декорация4
    - ГруппаНомера
      - Декорация5
      - Декорация6
      - Декорация7
      - Декорация8
  - ГруппаВводКапчи
    - ВведеныйКод (поле)
    - ПодтвердитьВвод (кноп)
  - ОтменаКапчи
    - ОтменитьВвод (кноп)
  - ГруппаРешенияКапчи
    - КапчаПройдена (надпис)
    - Декорация9
    - УспешныйВыход (кноп)
- Ошибка (надпис)
- Выход (кноп)

## Код

```1c
&НаКлиенте
Перем КапчаПройдена;
&НаКлиенте
Перем СчетчикОшибок;
&НаКлиенте
Перем ПравильныйОтвет;
&НаКлиенте
Перем КартинкиКапчи;

// Выбор формы для запуска прописан в Конфигурация -> Открыть модуль приложения
//!ВАЖНО! в свойствах формы РежимОткрытияОкна выбрать «Блокировать весь интерфейс», иначе капча не имеет смысла
&НаКлиенте
Процедура ПодтвердитьВвод(Команда)     
    Если СчетчикОшибок >= 2 Тогда // по условиям задания у нас есть 3 попытки
        ОбработкаОшибок("Капча не продена!");
        Элементы.ГруппаКапча.Видимость = Ложь;
        Элементы.ГруппаВводКапчи.Видимость = Ложь;
        Элементы.ОтменаКапчи.Видимость = Ложь;
        Элементы.Выход.Видимость = Истина;
    Иначе
        Если ПравильныйОтвет = СтрЗаменить(Строка(ВведенныйКод), " ", "") Тогда
            КапчаПройдена = Истина;
            Элементы.ГруппаРешениеКапчи.Видимость = Истина;
            Элементы.ГруппаКапча.Видимость = Ложь;
            Элементы.ГруппаВводКапчи.Видимость = Ложь;
            Элементы.ОтменаКапчи.Видимость = Ложь;
            Элементы.Выход.Видимость = Ложь;
            Элементы.Ошибка.Видимость = Ложь; 
            ОткрытьФорму("ОбщаяФорма.Авторизация");
        Иначе
            СчетчикОшибок = СчетчикОшибок + 1;
            ОбработкаОшибок("Неправильный ответ. Осталось попыток " + Строка(3 - СчетчикОшибок));
            ОбработкаКапчи();
        КонецЕсли;
    КонецЕсли;
КонецПроцедуры


&НаКлиенте
Процедура ПриОткрытии(Отказ)
    // Открываем на весь экран (опционально, просто чтобы не смущал интерфейс системы на заднем фоне)
    // WSHShell = Новый COMОбъект("WScript.Shell");
    // WSHShell.SendKeys("% ");
    // WSHShell.SendKeys("{DOWN}{DOWN}{DOWN}{DOWN}{ENTER}");
    СчетчикОшибок = 0;     
    КапчаПройдена = Ложь;
    ОбработкаКапчи();
КонецПроцедуры


&НаКлиенте
Процедура ПриЗакрытии(ЗавершениеРаботы)
    // В случае закрытия капчи через крестик, нас не впустит в систему, а завершит работу программы
    Если КапчаПройдена = Ложь Тогда
        ЗавершитьРаботуСистемы(Ложь, Ложь);
    КонецЕсли;
КонецПроцедуры

&НаКлиенте
Процедура ОбработкаКапчи()
    // Сохраняем изначальный порядок картинок
    //!ВАЖНО! в самих декорациях на форме нужно разместить картинки в правильном порядке,
    // от этого зависит какая последовательность будет считаться правильной
    Если КартинкиКапчи = Неопределено Тогда
        КартинкиКапчи = Новый Массив;
        КартинкиКапчи.Добавить(Элементы.Декорация1.Картинка); // фрагмент 1
        КартинкиКапчи.Добавить(Элементы.Декорация2.Картинка); // фрагмент 2
        КартинкиКапчи.Добавить(Элементы.Декорация3.Картинка); // фрагмент 3
        КартинкиКапчи.Добавить(Элементы.Декорация4.Картинка); // фрагмент 4
    КонецЕсли;

    // Записываем Декорации1‑4 для удобного обращения к ним
    СлотыКартинок = Новый Массив;
    СлотыКартинок.Добавить(Элементы.Декорация1);
    СлотыКартинок.Добавить(Элементы.Декорация2);
    СлотыКартинок.Добавить(Элементы.Декорация3);
    СлотыКартинок.Добавить(Элементы.Декорация4);

    // Записываем Декорации5‑8 для удобного обращения к ним
    СлотыЦифр = Новый Массив;
    СлотыЦифр.Добавить(Элементы.Декорация5);
    СлотыЦифр.Добавить(Элементы.Декорация6);
    СлотыЦифр.Добавить(Элементы.Декорация7);
    СлотыЦифр.Добавить(Элементы.Декорация8);

    ГСЧ = Новый ГенераторСлучайныхЧисел;

    // 1) Случайно раскладываем фрагменты по слотам
    ПорядокФрагментов = СлучайнаяПерестановка(4, ГСЧ);
    // 2) Случайно раскладываем цифры по слотам
    ЦифрыНаСлотах = СлучайнаяПерестановка(4, ГСЧ);

    // Показать пользователю
    Для i = 0 По 3 Цикл
        НомерФрагмента = ПорядокФрагментов[i];
        // В декорации по‑порядку подставляем сохранённые картинки в сгенерированной случайной последовательности
        СлотыКартинок[i].Картинка = КартинкиКапчи[НомерФрагмента - 1];
        // В декорации по‑порядку подставляем цифры
        СлотыЦифр[i].Заголовок = Строка(ЦифрыНаСлотах[i]);
    КонецЦикла;

    // 3) Считаем правильный ответ
    ПравильныйОтвет = "";
    Для Фрагмент = 1 По 4 Цикл
        ИндексСлота = ПорядокФрагментов.Найти(Фрагмент); // где сейчас стоит нужный фрагмент
        ПравильныйОтвет = ПравильныйОтвет + Строка(ЦифрыНаСлотах[ИндексСлота]);
    КонецЦикла;
     // Очищаем ввод
    ВведенныйКод = "";
КонецПроцедуры

Функция СлучайнаяПерестановка(Количество, ГСЧ)
    // Алгоритм Фишера‑Йетса, перемешивание чисел без их повтора
    М = Новый Массив;
    Для i = 1 По Количество Цикл
        М.Добавить(i);
    КонецЦикла;

    i = Количество;
    Пока i >= 2 Цикл
        j = ГСЧ.СлучайноеЧисло(1, i);
        Врем = М[i - 1];
        М[i - 1] = М[j - 1];
        М[j - 1] = Врем;
        i = i - 1;
    КонецЦикла;

    Возврат М;
КонецФункции

    // Отображаем сообщение об ошибке в надпись
    Элементы.Ошибка.Заголовок = Ошибка;
КонецФункции

    Закрыть();
```
